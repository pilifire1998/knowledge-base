## 原生模块开发

### 新架构：TurboModules (推荐)

React Native 0.71+ 支持新架构，使用 TurboModules 提供更好的类型安全和性能。

**1. 定义规范 (NativeSample.ts)**

```typescript
// src/NativeModules/NativeSample.ts
import type { TurboModule } from 'react-native';
import { TurboModuleRegistry } from 'react-native';

export interface Spec extends TurboModule {
  // 获取设备信息
  getDeviceName(): Promise<string>;

  // 计算操作
  add(a: number, b: number): Promise<number>;

  // 存储操作
  setItem(key: string, value: string): Promise<void>;
  getItem(key: string): Promise<string | null>;

  // 事件支持
  addListener(eventName: string): void;
  removeListeners(count: number): void;
}

export default TurboModuleRegistry.getEnforcing<Spec>('NativeSample');
```

**2. iOS 实现 (Objective-C++)**

```objc
// ios/NativeSample.mm
#import <React/RCTBridgeModule.h>
#import <React/RCTEventEmitter.h>
#import <ReactCommon/RCTTurboModule.h>

@interface NativeSample : RCTEventEmitter
@end

@implementation NativeSample

RCT_EXPORT_MODULE()

- (dispatch_queue_t)methodQueue {
  return dispatch_get_main_queue();
}

- (NSArray<NSString *> *)supportedEvents {
  return @[@"onCustomEvent"];
}

// 获取设备名称
RCT_REMAP_METHOD(getDeviceName,
                 getDeviceNameWithResolver:(RCTPromiseResolveBlock)resolve
                 rejecter:(RCTPromiseRejectBlock)reject) {
  NSString *deviceName = [[UIDevice currentDevice] name];
  resolve(deviceName);
}

// 加法计算
RCT_REMAP_METHOD(add,
                 addWithA:(double)a
                 b:(double)b
                 resolver:(RCTPromiseResolveBlock)resolve
                 rejecter:(RCTPromiseRejectBlock)reject) {
  resolve(@(a + b));
}

// 存储操作
RCT_REMAP_METHOD(setItem,
                 setItemWithKey:(NSString *)key
                 value:(NSString *)value
                 resolver:(RCTPromiseResolveBlock)resolve
                 rejecter:(RCTPromiseRejectBlock)reject) {
  NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];
  [defaults setObject:value forKey:key];
  [defaults synchronize];
  resolve(nil);
}

RCT_REMAP_METHOD(getItem,
                 getItemWithKey:(NSString *)key
                 resolver:(RCTPromiseResolveBlock)resolve
                 rejecter:(RCTPromiseRejectBlock)reject) {
  NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];
  NSString *value = [defaults stringForKey:key];
  resolve(value ?: [NSNull null]);
}

// 发送事件
- (void)sendCustomEvent:(NSString *)message {
  [self sendEventWithName:@"onCustomEvent" body:@{@"message": message}];
}

@end
```

**3. Android 实现 (Kotlin)**

```kotlin
// android/app/src/main/java/com/myapp/NativeSampleModule.kt
package com.myapp

import android.content.Context
import com.facebook.react.bridge.*
import com.facebook.react.modules.core.DeviceEventManagerModule

class NativeSampleModule(reactContext: ReactApplicationContext) :
    ReactContextBaseJavaModule(reactContext) {

    override fun getName(): String = "NativeSample"

    // 获取设备名称
    @ReactMethod
    fun getDeviceName(promise: Promise) {
        val deviceName = android.os.Build.MODEL
        promise.resolve(deviceName)
    }

    // 加法计算
    @ReactMethod
    fun add(a: Double, b: Double, promise: Promise) {
        promise.resolve(a + b)
    }

    // 存储操作
    @ReactMethod
    fun setItem(key: String, value: String, promise: Promise) {
        val prefs = reactApplicationContext.getSharedPreferences("NativeSample", Context.MODE_PRIVATE)
        prefs.edit().putString(key, value).apply()
        promise.resolve(null)
    }

    @ReactMethod
    fun getItem(key: String, promise: Promise) {
        val prefs = reactApplicationContext.getSharedPreferences("NativeSample", Context.MODE_PRIVATE)
        val value = prefs.getString(key, null)
        promise.resolve(value)
    }

    // 发送事件
    fun sendCustomEvent(message: String) {
        reactApplicationContext
            .getJSModule(DeviceEventManagerModule.RCTDeviceEventEmitter::class.java)
            .emit("onCustomEvent", Arguments.createMap().apply {
                putString("message", message)
            })
    }

    // 事件监听器注册
    @ReactMethod
    fun addListener(eventName: String) {
        // 占位符，用于注册事件监听
    }

    @ReactMethod
    fun removeListeners(count: Int) {
        // 占位符，用于移除事件监听
    }
}
```

**4. JavaScript 使用**

```tsx
// hooks/useNativeSample.ts
import { NativeEventEmitter, NativeModules } from 'react-native';
import { useEffect, useState } from 'react';

const { NativeSample } = NativeModules;

export const useDeviceName = () => {
  const [deviceName, setDeviceName] = useState<string>('');

  useEffect(() => {
    NativeSample.getDeviceName()
      .then(setDeviceName)
      .catch(console.error);
  }, []);

  return deviceName;
};

export const useNativeEvents = () => {
  const [events, setEvents] = useState<string[]>([]);

  useEffect(() => {
    const eventEmitter = new NativeEventEmitter(NativeSample);
    const subscription = eventEmitter.addListener('onCustomEvent', (event) => {
      setEvents((prev) => [...prev, event.message]);
    });

    return () => subscription.remove();
  }, []);

  return events;
};

// 使用示例
const MyComponent = () => {
  const deviceName = useDeviceName();
  const events = useNativeEvents();

  const handleAdd = async () => {
    const result = await NativeSample.add(10, 20);
    console.log('Result:', result); // 30
  };

  return (
    <View>
      <Text>设备: {deviceName}</Text>
      <Button title="计算 10 + 20" onPress={handleAdd} />
    </View>
  );
};
```

## 高级动画：Reanimated 3

### 基础动画

```tsx
import Animated, {
  useSharedValue,
  useAnimatedStyle,
  withTiming,
  withSpring,
  withRepeat,
  Easing,
} from 'react-native-reanimated';

const BasicAnimation = () => {
  const offset = useSharedValue(0);

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [{ translateX: offset.value }],
  }));

  const move = () => {
    // 线性动画
    offset.value = withTiming(100, { duration: 500 });

    // 弹簧动画
    offset.value = withSpring(100, {
      damping: 10,
      stiffness: 100,
    });

    // 重复动画
    offset.value = withRepeat(
      withTiming(100, { duration: 500 }),
      -1, // 无限重复
      true // 反向
    );
  };

  return (
    <View>
      <Animated.View
        style={[
          { width: 100, height: 100, backgroundColor: 'red' },
          animatedStyle,
        ]}
      />
      <Button title="移动" onPress={move} />
    </View>
  );
};
```

### 手势交互

```tsx
import { Gesture, GestureDetector } from 'react-native-gesture-handler';
import Animated, {
  useSharedValue,
  useAnimatedStyle,
  withSpring,
} from 'react-native-reanimated';

const DraggableCard = () => {
  const translateX = useSharedValue(0);
  const translateY = useSharedValue(0);
  const scale = useSharedValue(1);

  const panGesture = Gesture.Pan()
    .onUpdate((e) => {
      translateX.value = e.translationX;
      translateY.value = e.translationY;
    })
    .onEnd(() => {
      translateX.value = withSpring(0);
      translateY.value = withSpring(0);
    });

  const pinchGesture = Gesture.Pinch()
    .onUpdate((e) => {
      scale.value = e.scale;
    })
    .onEnd(() => {
      scale.value = withSpring(1);
    });

  const composed = Gesture.Simultaneous(panGesture, pinchGesture);

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [
      { translateX: translateX.value },
      { translateY: translateY.value },
      { scale: scale.value },
    ],
  }));

  return (
    <GestureDetector gesture={composed}>
      <Animated.View
        style={[
          {
            width: 150,
            height: 150,
            backgroundColor: '#007AFF',
            borderRadius: 12,
          },
          animatedStyle,
        ]}
      />
    </GestureDetector>
  );
};
```

### 自定义动画钩子

```tsx
// hooks/useFlipAnimation.ts
import { useSharedValue, useAnimatedStyle, withTiming } from 'react-native-reanimated';

export const useFlipAnimation = () => {
  const rotateY = useSharedValue(0);

  const frontStyle = useAnimatedStyle(() => ({
    transform: [
      { perspective: 1000 },
      { rotateY: `${rotateY.value}deg` },
    ],
    opacity: rotateY.value < 90 ? 1 : 0,
  }));

  const backStyle = useAnimatedStyle(() => ({
    transform: [
      { perspective: 1000 },
      { rotateY: `${rotateY.value + 180}deg` },
    ],
    opacity: rotateY.value >= 90 ? 1 : 0,
  }));

  const flip = () => {
    rotateY.value = withTiming(rotateY.value === 0 ? 180 : 0, { duration: 500 });
  };

  return { frontStyle, backStyle, flip };
};
```

## 测试策略

### Jest 单元测试

```tsx
// __tests__/components/Button.test.tsx
import { render, fireEvent } from '@testing-library/react-native';
import Button from '../components/Button';

describe('Button', () => {
  it('renders correctly', () => {
    const { getByText } = render(<Button title="Click me" onPress={() => {}} />);
    expect(getByText('Click me')).toBeTruthy();
  });

  it('calls onPress when pressed', () => {
    const onPressMock = jest.fn();
    const { getByText } = render(<Button title="Click" onPress={onPressMock} />);

    fireEvent.press(getByText('Click'));
    expect(onPressMock).toHaveBeenCalledTimes(1);
  });

  it('is disabled when disabled prop is true', () => {
    const onPressMock = jest.fn();
    const { getByText } = render(
      <Button title="Click" onPress={onPressMock} disabled />
    );

    fireEvent.press(getByText('Click'));
    expect(onPressMock).not.toHaveBeenCalled();
  });
});
```

```tsx
// __tests__/hooks/useCounter.test.ts
import { renderHook, act } from '@testing-library/react-hooks';
import { useCounter } from '../hooks/useCounter';

describe('useCounter', () => {
  it('initializes with default value', () => {
    const { result } = renderHook(() => useCounter());
    expect(result.current.count).toBe(0);
  });

  it('increments count', () => {
    const { result } = renderHook(() => useCounter());

    act(() => {
      result.current.increment();
    });

    expect(result.current.count).toBe(1);
  });
});
```
